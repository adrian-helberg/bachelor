{

    frame=tB,

    basicstyle=\scriptsize,
    keywordstyle=\color{black}\bfseries\em,

    xleftmargin=.04\textwidth,
    #1 % this is to add specific settings to an usage of this environment (for instnce, the caption and referable label)
  }

      {Ä}{{\"A}}1
